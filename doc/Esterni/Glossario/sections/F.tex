	\parola{Fetch}{%
		Recupero delle informazioni da una o più \gloss{repository} che aggiorna lo stato dei branch remoti tracciati.
	}\label{Fetch}
	
	\parola{Firebase}{%
		Piattaforma mobile di Google di aiuto per lo sviluppo veloce di app di alta qualità.
	}\label{Firebase}

	\parola{Formazione}{
		Progressiva acquisizione, attraverso lo studio o l'esperienza, di una determinata fisionomia culturale o morale, di competenze specifiche. È sinonimo di preparazione, educazione.
	}\label{Formazione}

	\parola{Fornitore}{
		Fornitore è un soggetto economico che opera nel mercato d'approvvigionamento per le risorse finanziarie e primarie (beni e servizi). 
	}\label{Fornitore}

	\parola{Framework}{
		Struttura che funge da strato intermedio tra un sistema operativo e il software che lo utilizza. È un'architettura logica di supporto che spesso facilita al programmatore la realizzazione del software.  
	}\label{Framework}
