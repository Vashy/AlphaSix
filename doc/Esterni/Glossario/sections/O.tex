    \parola{Octalysis}{%
        \`E un progetto che cerca di ottimizzare le scelte umane servendosi del coinvolgimento, degli interessi e della
        motivazione.
    }\label{Octalysis}
	
	\parola{Open source}{%
		Viene utilizzato per riferirsi ad un software di cui i detentori dei diritti rendono pubblico il codice
		sorgente, favorendone il libero studio e permettendo a programmatori indipendenti di apportarvi modifiche
		ed estensioni.
	}\label{Open source}
	
	\parola{Operation}{%
		Insieme di fasi finalizzate a mettere a disposizione il prodotto finale all'utente. Coinvolgono dunque la logistica, la progettazione e lo sviluppo.
	}\label{Operation}
