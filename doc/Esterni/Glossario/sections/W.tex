\lettera{W}\label{W}

	\parola{Way of working}{%
		Modo di lavorare caratterizzato dalle best practice scelte. Esso non potrà per forza essere corretto fin dall'inizio del progetto, ma verrà raffinato nel tempo attraverso la manutenzione migliorativa.
	}\label{Way of working}

	\parola{Webhook}{
		Metodo per aumentare o modificare il comportamento di una pagina o applicazione web con chiamate di ritorno personalizzate (\gloss{callback}).
	}\label{Webhook}

	\parola{Workflow}{%
		Rappresenta un insieme di compiti e di diversi attori coinvolti nella realizzazione di un processo
		lavorativo.\\
		Nel caso di \gloss{Amazon Alexa} si riferisce ad un insieme di \gloss{Skill}, le quali vengono eseguite in sequenza
		senza il bisogno di attivarle singolarmente.
	}\label{Workflow}
	
    \parola{Workspace}{%
        Termine inglese per "spazio di lavoro".
    }\label{Workspace}
